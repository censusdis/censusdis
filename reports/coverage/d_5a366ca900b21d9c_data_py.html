46GeoFilterType = Optional[Union[str, Iterable[str]]] 

47""" 

48The type we accept for geographic filters. 

49 

50They are used for the values of `kwargs` to 

51:py:func:`download`. 

52 

53These filters are either single values as a string, 

54or, if multivalued, then an iterable containing all 

55the values allowed by the filter. For example:: 

56 

57 import censusdis.data as ced 

58 

59 from censusdis.states import NJ, NY, CT 

60 

61 # Two different kinds of kwarg for `state=`, both of 

62 # which are of `GeoFilterType`: 

63 df_one_state = ced.download("aca/acs5", 2020, ["NAME"], state=NJ) 

64 df_tri_state = ced.download("aca/acs5", 2020, ["NAME"], state=[NJ, NY, CT]) 

65"""

80_MAX_VARIABLES_PER_DOWNLOAD = 50 

81""" 

82The maximum number of variables we can ask for in one census API query. 

83 

84The U.S. Census sets this limit, not us. In order to not expose our 

85users to the limit, :py:func:`~download` mostly obscures the fact that 

86requests to download more than this many variables are broken into 

87multiple calls to the census API and then the results are stitched back 

88together be either merging or concatenation. This is all handled in 

89:py:func:`~_download_multiple`. 

90"""

111def _download_multiple( 

112 dataset: str, 

113 vintage: VintageType, 

114 download_variables: List[str], 

115 *, 

116 key: Optional[str], 

117 census_variables: "VariableCache", 

118 with_geometry: bool = False, 

119 row_keys: Optional[Union[str, Iterable[str]]] = None, 

120 **kwargs: cgeo.InSpecType, 

121) -> pd.DataFrame: 

122 """ 

123 Download data in groups of columns and concatenate the results together. 

124 

125 The reason for this function is that the API will only return a maximum 

126 of 50 columns per query. This function downloads wider data 50 columns 

127 at a time and concatenates them. 

128 

129 Parameters 

130 ---------- 

131 dataset 

132 The dataset to download from. For example `"acs/acs5"`, 

133 `"dec/pl"`, or `"timeseries/poverty/saipe/schdist"`. 

134 vintage 

135 The vintage to download data for. For most data sets this is 

136 an integer year, for example, `2020`. But for 

137 a timeseries data set, pass the string `'timeseries'`. 

138 download_variables 

139 The census variables to download, for example `["NAME", "B01001_001E"]`. 

140 with_geometry 

141 If `True` a :py:class:`gpd.GeoDataFrame` will be returned and each row 

142 will have a geometry that is a cartographic boundary suitable for platting 

143 a map. See https://www.census.gov/geographies/mapping-files/time-series/geo/cartographic-boundary.2020.html 

144 for details of the shapefiles that will be downloaded on your behalf to 

145 generate these boundaries. 

146 api_key 

147 An optional API key. If you don't have or don't use a key, the number 

148 of calls you can make will be limited. 

149 variable_cache 

150 A cache of metadata about variables. 

151 row_keys 

152 An optional set of identifier keys to help merge together requests for more than the census API limit of 

153 50 variables per query. These keys are useful for census datasets such as the Current Population Survey 

154 where the geographic identifiers do not uniquely identify each row. 

155 kwargs 

156 A specification of the geometry that we want data for. 

157 

158 Returns 

159 ------- 

160 The full results of the query with all columns. 

161 

162 """ 

163 # Divide the variables into groups. If row keys are provided, include them in each chunk of variables, 

164 # while respecting the variable max 

165 if row_keys: 

166 chunk_size = _MAX_VARIABLES_PER_DOWNLOAD - len(row_keys) 

167 variable_groups = [ 

168 # black and flake8 disagree about the whitespace before ':' here... 

169 # We need to drop duplicates in each chunk of variables 

170 # since the row_key variables might already be present in one of the chunks 

171 [ 

172 item 

173 for item in row_keys 

174 + download_variables[start : start + chunk_size] # noqa: E203 

175 if item not in row_keys 

176 or row_keys.index(item) 

177 == ( 

178 row_keys 

179 + download_variables[start : start + chunk_size] # noqa: E203 

180 ).index(item) 

181 ] 

182 for start in range(0, len(download_variables), chunk_size) 

183 ] 

184 else: 

185 variable_groups = [ 

186 # black and flake8 disagree about the whitespace before ':' here... 

187 download_variables[start : start + _MAX_VARIABLES_PER_DOWNLOAD] # noqa: 203 

188 for start in range(0, len(download_variables), _MAX_VARIABLES_PER_DOWNLOAD) 

189 ] 

190 

191 if len(variable_groups) < 2: 

192 raise ValueError( 

193 "_download_multiple expects to be called with at least " 

194 f"{_MAX_VARIABLES_PER_DOWNLOAD + 1} variables. With fewer," 

195 "use download instead." 

196 ) 

197 

198 # Get the data for each chunk. 

199 dfs = [ 

200 download( 

201 dataset, 

202 vintage, 

203 variable_group, 

204 api_key=key, 

205 variable_cache=census_variables, 

206 with_geometry=with_geometry and (ii == 0), 

207 **kwargs, 

208 ) 

209 for ii, variable_group in enumerate(variable_groups) 

210 ] 

211 

212 # What variables came back in the first df but were not 

213 # requested? These are a key to the geography the row 

214 # represents. For example, 'STATE' amd 'COUNTY' might 

215 # be these variables if we did a county-level query to 

216 # the census API. 

217 geo_key_variables = [f for f in dfs[0].columns if f not in set(variable_groups[0])] 

218 

219 # If we put in the geometry column, it's not part of the 

220 # key. 

221 if with_geometry: 

222 geo_key_variables = [f for f in geo_key_variables if f != "geometry"] 

223 

224 # Now we have to decide if we are going to use the merge 

225 # strategy or the concat strategy to combine the data frames 

226 # we downloaded. Why do we have two strategies? Because we are 

227 # dealing with two kinds of data. One kind, from data sets like 

228 # ACS (https://www.census.gov/programs-surveys/acs.html), 

229 # has a unique key of columns that specify geography. The other 

230 # kind, from data sets like CPS 

231 # (https://www.census.gov/programs-surveys/cps.html) doesn't. 

232 # 

233 # In the unique key case, we can join the data frames that come 

234 # back on those key columns and get the final wide data frame 

235 # we want for the user. 

236 # 

237 # In the non-unique key case, we can't do this. There data sets 

238 # may have multiple rows for a value of the key columns. We can't 

239 # join here. Instead, we can only concatenate the tables 

240 # horizontally and hope that the rows came back in the same order 

241 # for each of them. 

242 

243 # We hope to be able to merge. It is safer. If row_keys is supplied, they are included in 

244 # merge keys 

245 merge_strategy = True 

246 if row_keys: 

247 merge_keys = geo_key_variables + row_keys 

248 else: 

249 merge_keys = geo_key_variables 

250 # But if there are any non-unique keys in any df, we can't 

251 # merge. 

252 for df_slice in dfs: 

253 if len(df_slice.value_counts(merge_keys, sort=False)) != len(df_slice.index): 

254 merge_strategy = False 

255 break 

256 

257 if merge_strategy: 

258 # We can do the merge strategy. 

259 

260 __dw_strategy_metrics["merge"] = __dw_strategy_metrics["merge"] + 1 

261 

262 df_data = dfs[0] 

263 

264 for df_right in dfs[1:]: 

265 df_data = df_data.merge(df_right, on=merge_keys) 

266 else: 

267 # We are going to have to fall back on the concat 

268 # strategy. Before we do the concat, however, let's 

269 # double-check that the key columns are the same in 

270 # at the corresponding row in every df. Otherwise, something 

271 # is fishy, and it is not safe to concat without mixing 

272 # data that should be in different rows. 

273 

274 rows0 = len(dfs[0].index) 

275 

276 for df_slice in dfs[1:]: 

277 if not ( 

278 rows0 == len(df_slice.index) 

279 and dfs[0][geo_key_variables].equals(df_slice[geo_key_variables]) 

280 ): 

281 # At least one difference. So we cannot use the 

282 # concat strategy either. 

283 if not row_keys: 

284 raise CensusApiException( 

285 "Neither the merge nor the concat strategy is viable. " 

286 "We made multiple queries to the census API because more than " 

287 f"{_MAX_VARIABLES_PER_DOWNLOAD} variables were requested. " 

288 "If you don't need all the variables, it is always safer to " 

289 f"download less than {_MAX_VARIABLES_PER_DOWNLOAD} variables. " 

290 f"If you need more than {_MAX_VARIABLES_PER_DOWNLOAD}, you can supply the `row_keys`" 

291 "arguement with a set of variables that uniquely identify each row." 

292 ) 

293 else: 

294 raise CensusApiException( 

295 f"Neither the merge nor the concat strategy is viable using row_keys: {row_keys}. " 

296 "The supplied keys should uniquely identify every row in the dataset to work. " 

297 "If you don't need all the variables, it is always safer to " 

298 f"download less than {_MAX_VARIABLES_PER_DOWNLOAD} variables. " 

299 ) 

300 

301 # Concat strategy is as safe as it will ever be. We hope the server 

302 # side did not reorder the results across queries. 

303 logger.info( 

304 "Using the concat strategy, which is not guaranteed reliable if " 

305 "the census API returned data for multiple sub-queries of less than " 

306 "or equal to %d in different row orders. " 

307 "It is always safest to query no more than %d " 

308 "variables at a time. Please do so unless you really need them all.", 

309 _MAX_VARIABLES_PER_DOWNLOAD, 

310 _MAX_VARIABLES_PER_DOWNLOAD, 

311 ) 

312 

313 __dw_strategy_metrics["concat"] = __dw_strategy_metrics["concat"] + 1 

314 

315 df_data = pd.concat( 

316 [dfs[0]] + [df.drop(geo_key_variables, axis="columns") for df in dfs[1:]], 

317 axis="columns", 

318 ) 

319 

320 return df_data

323def download( 

324 dataset: str, 

325 vintage: VintageType, 

326 download_variables: Optional[Union[str, Iterable[str]]] = None, 

327 *, 

328 group: Optional[Union[str, Iterable[str]]] = None, 

329 leaves_of_group: Optional[Union[str, Iterable[str]]] = None, 

330 set_to_nan: Union[bool, Iterable[int]] = True, 

331 skip_annotations: bool = True, 

332 with_geometry: bool = False, 

333 remove_water: bool = False, 

334 download_contained_within: Optional[Dict[str, cgeo.InSpecType]] = None, 

335 area_threshold: float = 0.8, 

336 api_key: Optional[str] = None, 

337 variable_cache: Optional["VariableCache"] = None, 

338 row_keys: Optional[Union[str, Iterable[str]]] = None, 

339 **kwargs: cgeo.InSpecType, 

340) -> Union[pd.DataFrame, gpd.GeoDataFrame]: 

341 """ 

342 Download data from the US Census API. 

343 

344 This is the main API for downloading US Census data with the 

345 `censusdis` package. There are many examples of how to use 

346 this in the demo notebooks provided with the package at 

347 https://github.com/vengroff/censusdis/tree/main/notebooks. 

348 

349 *A note on variables and groups*: there are multiple ways to specify the 

350 variables you want to download, either individually in `download_variables`, 

351 by one or more groups in `group`, and by the leaves of one or more groups 

352 in `leaves_of_group`. Note that these three sources af variables are 

353 deduplicated, so you will only get one column for a variable no matter 

354 how many times it is specified. 

355 

356 *Specifying census geographies*: censusdis provides access to many 

357 census datasets, each of which can be retrieved at a particular set of 

358 geographic grains. To accomodate this, `download()` takes a set 

359 of kwargs to define the geographic level of the returned data. You can check 

360 which geographies are available for a particular dataset with the 

361 `geographies()`. 

362 

363 Parameters 

364 ---------- 

365 dataset 

366 The dataset to download from. For example `"acs/acs5"`, 

367 `"dec/pl"`, or `"timeseries/poverty/saipe/schdist"`. There are 

368 symbolic names for datasets, like `ACS5` for `"acs/acs5" 

369 in :py:module:`censusdis.datasets`. 

370 vintage 

371 The vintage to download data for. For most data sets this is 

372 an integer year, for example, `2020`. But for 

373 a timeseries data set, pass the string `'timeseries'`. 

374 download_variables 

375 The census variables to download, for example `["NAME", "B01001_001E"]`. 

376 group 

377 One or more groups (as defined by the U.S. Census for the data set) 

378 whose variable values should be downloaded. These are in addition to 

379 any specified in `download_variables`. 

380 leaves_of_group 

381 One or more groups (as defined by the U.S. Census for the data set) 

382 whose leaf variable values should be downloaded.These are in addition to 

383 any specified in `download_variables` or `group`. See 

384 :py:meth:`VariableCache.group_leaves` for more details on the semantics of 

385 leaves vs. non-leaf group variables. 

386 set_to_nan 

387 A list of values that should be set to NaN. Normally these are special 

388 values that the U.S. Census API sometimes returns. If `True`, then all 

389 values in :py:ref:`censusdis.values.ALL_SPECIAL_VALUES` will be replaced. 

390 If `False`, no replacements will be made. 

391 skip_annotations 

392 If `True` try to filter out `group` or `leaves_of_group` variables that are 

393 annotations rather than actual values. See :py:meth:`VariableCache.group_variables` 

394 for more details. Variable names passed in `download_variables` are not 

395 affected by this flag. 

396 with_geometry 

397 If `True` a :py:class:`gpd.GeoDataFrame` will be returned and each row 

398 will have a geometry that is a cartographic boundary suitable for platting 

399 a map. See https://www.census.gov/geographies/mapping-files/time-series/geo/cartographic-boundary.2020.html 

400 for details of the shapefiles that will be downloaded on your behalf to 

401 generate these boundaries. 

402 remove_water 

403 If `True` and if with_geometry=True, will query TIGER for AREAWATER shapefiles and 

404 remove water areas from returned geometry. 

405 download_contained_within 

406 A dictionary specifying the geography or geographies that our results 

407 should be filtered down to be contained within. 

408 area_threshold 

409 What fraction of the area of other geographies must be contained 

410 in our geography to be included. Ignored if `download_contained_within` is 

411 `None`. 

412 api_key 

413 An optional API key. If you don't have or don't use a key, the number 

414 of calls you can make will be limited to 500 per day. 

415 variable_cache 

416 A cache of metadata about variables. 

417 row_keys 

418 An optional set of identifier keys to help merge together requests for more than the census API limit of 

419 50 variables per query. These keys are useful for census datasets such as the Current Population Survey 

420 where the geographic identifiers do not uniquely identify each row. 

421 kwargs 

422 A specification of the geometry that we want data for. For example, 

423 `state = "*", county = "*"` will download county-level data for 

424 the entire US. 

425 

426 Returns 

427 ------- 

428 A :py:class:`~pd.DataFrame` or `~gpd.GeoDataFrame` containing the requested US Census data. 

429 """ 

430 if download_contained_within is not None: 

431 # Put the contained_within context around it. 

432 return contained_within( 

433 area_threshold=area_threshold, **download_contained_within 

434 ).download( 

435 dataset, 

436 vintage, 

437 download_variables, 

438 group=group, 

439 leaves_of_group=leaves_of_group, 

440 set_to_nan=set_to_nan, 

441 skip_annotations=skip_annotations, 

442 with_geometry=with_geometry, 

443 remove_water=remove_water, 

444 api_key=api_key, 

445 variable_cache=variable_cache, 

446 row_keys=row_keys, 

447 **kwargs, 

448 ) 

449 

450 if variable_cache is None: 

451 variable_cache = variables 

452 

453 # Ensure list operations work 

454 if row_keys: 

455 row_keys = list(row_keys) 

456 

457 # The side effect here is to prime the cache. 

458 cgeo.geo_path_snake_specs(dataset, vintage) 

459 

460 if set_to_nan is True: 

461 set_to_nan = ALL_SPECIAL_VALUES 

462 

463 # In case they came to us in py format, as kwargs often do. 

464 kwargs = { 

465 cgeo.path_component_from_snake(dataset, vintage, k): v 

466 for k, v in kwargs.items() 

467 } 

468 

469 # Parse out the download variables 

470 download_variables = _parse_download_variables( 

471 dataset, 

472 vintage, 

473 download_variables=download_variables, 

474 group=group, 

475 leaves_of_group=leaves_of_group, 

476 skip_annotations=skip_annotations, 

477 variable_cache=variable_cache, 

478 ) 

479 

480 if len(download_variables) <= _MAX_VARIABLES_PER_DOWNLOAD and row_keys: 

481 warnings.warn( 

482 "\n The row_keys arguement is intended to be used only when the number of requested" 

483 "\n variables exceeds the Census defined limit of 50" 

484 "\n The supplied value(s) will be ignored", 

485 UserWarning, 

486 ) 

487 # Special case if we are trying to get too many fields. 

488 if len(download_variables) > _MAX_VARIABLES_PER_DOWNLOAD: 

489 return _download_multiple( 

490 dataset, 

491 vintage, 

492 download_variables, 

493 key=api_key, 

494 census_variables=variable_cache, 

495 with_geometry=with_geometry, 

496 row_keys=row_keys, 

497 **kwargs, 

498 ) 

499 

500 # Prefetch all the types before we load the data. 

501 # That way we fail fast if a field is not known. 

502 _prefetch_variable_types(dataset, vintage, download_variables, variable_cache) 

503 # Also check that the row_keys, if supplied, are present in the dataset 

504 if row_keys: 

505 _prefetch_variable_types(dataset, vintage, row_keys, variable_cache) 

506 

507 # If we were given a list, join it together into 

508 # a comma-separated string. 

509 string_kwargs = {k: _gf2s(v) for k, v in kwargs.items()} 

510 

511 return _download_remote( 

512 dataset, 

513 vintage, 

514 download_variables=download_variables, 

515 set_to_nan=set_to_nan, 

516 with_geometry=with_geometry, 

517 remove_water=remove_water, 

518 api_key=api_key, 

519 variable_cache=variable_cache, 

520 **string_kwargs, 

521 )

524def _download_remote( 

525 dataset: str, 

526 vintage: VintageType, 

527 *, 

528 download_variables: List[str], 

529 set_to_nan: Union[bool, Iterable[float]] = True, 

530 with_geometry: bool, 

531 remove_water: bool, 

532 api_key: Optional[str], 

533 variable_cache: "VariableCache", 

534 **kwargs, 

535) -> Union[pd.DataFrame, gpd.GeoDataFrame]: 

536 """ 

537 Make the actual remote call to download the data. 

538 

539 This is the final step after we have parsed out and 

540 validated the variables and geometry. 

541 

542 Parameters 

543 ---------- 

544 dataset 

545 The dataset to download from. For example `"acs/acs5"`, 

546 `"dec/pl"`, or `"timeseries/poverty/saipe/schdist"`. 

547 vintage 

548 The vintage to download data for. For most data sets this is 

549 an integer year, for example, `2020`. But for 

550 a timeseries data set, pass the string `'timeseries'`. 

551 download_variables 

552 The census variables to download, for example `["NAME", "B01001_001E"]`. 

553 set_to_nan 

554 A list of values that should be set to NaN. Normally these are special 

555 values that the U.S. Census API sometimes returns. If `True`, then all 

556 values in :py:ref:`censusdis.values.ALL_SPECIAL_VALUES` will be replaced. 

557 If `False`, no replacements will be made. 

558 with_geometry 

559 If `True` a :py:class:`gpd.GeoDataFrame` will be returned and each row 

560 will have a geometry that is a cartographic boundary suitable for platting 

561 a map. See https://www.census.gov/geographies/mapping-files/time-series/geo/cartographic-boundary.2020.html 

562 for details of the shapefiles that will be downloaded on your behalf to 

563 generate these boundaries. 

564 api_key 

565 An optional API key. If you don't have or don't use a key, the number 

566 of calls you can make will be limited. 

567 variable_cache 

568 A cache of metadata about variables. 

569 kwargs 

570 A specification of the geometry that we want data for. 

571 

572 Returns 

573 ------- 

574 The downloaded variables, with or without added geometry, as 

575 either a `pd.DataFrame` or `gpd.GeoDataFrame`. 

576 """ 

577 url, params, bound_path = census_table_url( 

578 dataset, vintage, download_variables, api_key=api_key, **kwargs 

579 ) 

580 df_data = data_from_url(url, params) 

581 

582 # Coerce the types based on metadata about the variables. 

583 _coerce_downloaded_variable_types( 

584 dataset, vintage, download_variables, df_data, variable_cache 

585 ) 

586 

587 download_variables_upper = [dv.upper() for dv in download_variables] 

588 

589 # Put the geo fields (STATE, COUNTY, etc...) that came back up front. 

590 df_data = df_data[ 

591 [col for col in df_data.columns if col not in download_variables_upper] 

592 + download_variables_upper 

593 ] 

594 

595 # NaN out as requested. 

596 if set_to_nan is True: 

597 set_to_nan = ALL_SPECIAL_VALUES 

598 if set_to_nan: 

599 df_data = df_data.replace(list(set_to_nan), np.nan) 

600 

601 if with_geometry: 

602 # We need to get the geometry and merge it in. 

603 geo_level = bound_path.path_spec.path[-1] 

604 shapefile_scope = bound_path.bindings[bound_path.path_spec.path[0]] 

605 

606 gdf_data = add_geography(df_data, vintage, shapefile_scope, geo_level) 

607 

608 if remove_water: 

609 gdf_data = clip_water(gdf_data, vintage) 

610 

611 return gdf_data 

612 

613 return df_data

616def _coerce_downloaded_variable_types( 

617 dataset: str, 

618 vintage: VintageType, 

619 download_variables: List[str], 

620 df_data: pd.DataFrame, 

621 variable_cache: "VariableCache", 

622) -> None: 

623 """ 

624 Coerce the type of each returned variable (column) in a data frame. 

625 

626 We look up the type in the metadata in `variable_cache`. 

627 

628 Parameters 

629 ---------- 

630 dataset 

631 The dataset to download from. For example `"acs/acs5"`, 

632 `"dec/pl"`, or `"timeseries/poverty/saipe/schdist"`. 

633 vintage 

634 The vintage to download data for. For most data sets this is 

635 an integer year, for example, `2020`. But for 

636 a timeseries data set, pass the string `'timeseries'`. 

637 download_variables 

638 The census variables to download, for example `["NAME", "B01001_001E"]`. 

639 df_data 

640 The data that came back in JSON form from the census API. 

641 variable_cache 

642 A cache of metadata about variables. 

643 """ 

644 for variable in download_variables: 

645 # predicateType does not exist in some older data sets like acs/acs3 

646 # So in that case we just go with what we got in the JSON. But if we 

647 # have it try to set the type. 

648 if "predicateType" in variable_cache.get(dataset, vintage, variable): 

649 field_type = variable_cache.get(dataset, vintage, variable)["predicateType"] 

650 

651 if field_type == "int" or field_type == "long": 

652 if df_data[variable].isnull().any(): 

653 # Some Census data sets put in null in int fields. 

654 # We have to go with a float to make this a NaN. 

655 # Int has no representation for NaN or None. 

656 df_data[variable] = df_data[variable].astype(float, errors="ignore") 

657 else: 

658 try: 

659 df_data[variable] = df_data[variable].astype(int) 

660 except ValueError: 

661 # Sometimes census metadata says int, but they 

662 # put in float values anyway, so fall back on 

663 # trying to get them as floats. 

664 df_data[variable] = df_data[variable].astype( 

665 float, errors="ignore" 

666 ) 

667 except OverflowError: 

668 # Some long IDs are actually better handled as strings. 

669 df_data[variable] = df_data[variable].astype(str) 

670 elif field_type == "float": 

671 df_data[variable] = df_data[variable].astype(float) 

672 elif field_type == "string": 

673 pass 

674 else: 

675 # Leave it as an object? 

676 pass

726def _parse_download_variables( 

727 dataset: str, 

728 vintage: VintageType, 

729 *, 

730 download_variables: Optional[Union[str, Iterable[str]]] = None, 

731 group: Optional[Union[str, Iterable[str]]] = None, 

732 leaves_of_group: Optional[Union[str, Iterable[str]]] = None, 

733 skip_annotations: bool = True, 

734 variable_cache: Optional["VariableCache"] = None, 

735) -> List[str]: 

736 """ 

737 Parse out the full set of download variables. 

738 

739 These may be encoded in `download_variables`, `group`, and/or `leaves_of_group`. 

740 

741 Parameters 

742 ---------- 

743 dataset 

744 The dataset to download from. For example `"acs/acs5"`, 

745 `"dec/pl"`, or `"timeseries/poverty/saipe/schdist"`. 

746 vintage 

747 The vintage to download data for. For most data sets this is 

748 an integer year, for example, `2020`. But for 

749 a timeseries data set, pass the string `'timeseries'`. 

750 download_variables 

751 The census variables to download, for example `["NAME", "B01001_001E"]`. 

752 group 

753 One or more groups (as defined by the U.S. Census for the data set) 

754 whose variable values should be downloaded. These are in addition to 

755 any specified in `download_variables`. 

756 leaves_of_group 

757 One or more groups (as defined by the U.S. Census for the data set) 

758 whose leaf variable values should be downloaded.These are in addition to 

759 any specified in `download_variables` or `group`. See 

760 :py:meth:`VariableCache.group_leaves` for more details on the semantics of 

761 leaves vs. non-leaf group variables. 

762 skip_annotations 

763 If `True` try to filter out `group` or `leaves_of_group` variables that are 

764 annotations rather than actual values. See :py:meth:`VariableCache.group_variables` 

765 for more details. Variable names passed in `download_variables` are not 

766 affected by this flag. 

767 variable_cache 

768 A cache of metadata about variables. 

769 

770 Returns 

771 ------- 

772 The fully expanded list of variables to download. 

773 """ 

774 # Turn the variables we were given into a list if they are not already. 

775 if download_variables is None: 

776 download_variables = [] 

777 elif isinstance(download_variables, str): 

778 download_variables = [download_variables] 

779 elif not isinstance(download_variables, list): 

780 download_variables = list(download_variables) 

781 

782 if group is None: 

783 group = [] 

784 elif isinstance(group, str): 

785 group = [group] 

786 

787 if leaves_of_group is None: 

788 leaves_of_group = [] 

789 elif isinstance(leaves_of_group, str): 

790 leaves_of_group = [leaves_of_group] 

791 

792 # Add group variables and leaves as appropriate. 

793 group_variables: List[str] = [] 

794 for group_name in group: 

795 group_variables = group_variables + variable_cache.group_variables( 

796 dataset, vintage, group_name, skip_annotations=skip_annotations 

797 ) 

798 group_leaf_variables: List[str] = [] 

799 for group_name in leaves_of_group: 

800 group_leaf_variables = group_leaf_variables + variable_cache.group_leaves( 

801 dataset, vintage, group_name, skip_annotations=skip_annotations 

802 ) 

803 

804 # Concatenate them all. 

805 download_variables = download_variables + group_variables + group_leaf_variables 

806 

807 # Dedup and maintain order. 

808 download_variables = list(dict.fromkeys(download_variables)) 

809 

810 return download_variables

940def _intersecting_geos_kws( 

941 dataset: str, 

942 vintage: VintageType, 

943 containing_geo_kwargs: cgeo.InSpecType, 

944 **kwargs: cgeo.InSpecType, 

945) -> cgeo.InSpecType: 

946 """ 

947 Construct geography keywords for intersecting geographies. 

948 

949 Parameters 

950 ---------- 

951 dataset 

952 The dataset to download from. For example `"acs/acs5"`, 

953 `"dec/pl"`, or `"timeseries/poverty/saipe/schdist"`. There are 

954 symbolic names for datasets, like `ACS5` for `"acs/acs5" 

955 in :py:module:`censusdis.datasets`. 

956 vintage 

957 The vintage to download data for. For most data sets this is 

958 an integer year, for example, `2020`. But for 

959 a timeseries data set, pass the string `'timeseries'`. 

960 containing_geo_kwargs 

961 Geographic keywords specifying the containing geography that we are 

962 looking for intersections with. For example 

963 `dict(metropolitan_statistical_area_micropolitan_statistical_area="35620")` 

964 for the New York area CBSA. 

965 kwargs 

966 A specification of the geometry that we want data for, limited to those 

967 geographies that are contained in the geography specified by `containing_geo_kwargs`. 

968 For example, `state="*", county="*", tract="*"` will specifies county-level data for 

969 all counties contained in the containing geography. 

970 

971 Returns 

972 ------- 

973 A dictionary of geographic keywords suitable for passing to :py:func:`~download`. 

974 """ 

975 # This is a fast short circuit if there is only one 

976 # element of kwargs or the first component 

977 # is already specified. The former is since we will have to 

978 # query with the kwargs as they are, so we might as well 

979 # just let our caller do it. The second case is because 

980 # we might trim down the list, but more likely the user 

981 # double specified at the top level, like state=. 

982 if len(kwargs) == 1 or list(kwargs.values())[0] != "*": 

983 return kwargs 

984 

985 # Download the geometry of the outer scope. 

986 gdf_within = download( 

987 dataset, vintage, ["NAME"], with_geometry=True, **containing_geo_kwargs 

988 ) 

989 

990 # See if we can find a matching path spec. 

991 bound_path = _bind_path_if_possible(dataset, vintage, **kwargs) 

992 

993 # Get the geography for the outermost level of the match. 

994 first_binding = list(bound_path.bindings.items())[0] 

995 

996 containing_geo_kwargs = {first_binding[0]: first_binding[1]} 

997 

998 gdf_first_binding = download( 

999 dataset, vintage, ["NAME"], with_geometry=True, **containing_geo_kwargs 

1000 ) 

1001 

1002 # Which of the first binding geographies intersect 

1003 # the area we want our final geographies to be in. 

1004 gdf_intersects = gdf_first_binding.sjoin( 

1005 gdf_within, lsuffix="FIRST", rsuffix="within" 

1006 ) 

1007 

1008 col_name = first_binding[0].replace(" ", "_").upper() 

1009 if col_name not in gdf_intersects.columns: 

1010 col_name = f"{col_name}_FIRST" 

1011 

1012 intersecting_geographies = list(gdf_intersects[col_name].unique()) 

1013 

1014 # Short circuit if there are a massive number of intersection 

1015 # geps. In this case, we'll just leave things as they came with 

1016 # the leading '*' and query them all. Otherwise the URL gets super 

1017 # long and things go a little crazy. This can happen with zip code 

1018 # tabulation areas. 

1019 if len(intersecting_geographies) > 20: 

1020 return dict(**kwargs) 

1021 

1022 intersecting_geographies = [ 

1023 geo[:-6] if geo.endswith("_FIRST") else geo for geo in intersecting_geographies 

1024 ] 

1025 

1026 geo = dict(bound_path.bindings) 

1027 

1028 geo[first_binding[0]] = intersecting_geographies 

1029 

1030 return geo

1077 def download( 

1078 self, 

1079 dataset: str, 

1080 vintage: VintageType, 

1081 download_variables: Optional[Union[str, Iterable[str]]] = None, 

1082 *, 

1083 group: Optional[Union[str, Iterable[str]]] = None, 

1084 leaves_of_group: Optional[Union[str, Iterable[str]]] = None, 

1085 set_to_nan: Union[bool, Iterable[int]] = True, 

1086 skip_annotations: bool = True, 

1087 with_geometry: bool = False, 

1088 remove_water: bool = False, 

1089 api_key: Optional[str] = None, 

1090 variable_cache: Optional["VariableCache"] = None, 

1091 row_keys: Optional[Union[str, Iterable[str]]] = None, 

1092 **kwargs: cgeo.InSpecType, 

1093 ) -> Union[pd.DataFrame, gpd.GeoDataFrame]: 

1094 """ 

1095 Download data for geographies contained within a containing geography. 

1096 

1097 Parameters 

1098 ---------- 

1099 dataset 

1100 The dataset to download from. For example `"acs/acs5"`, 

1101 `"dec/pl"`, or `"timeseries/poverty/saipe/schdist"`. There are 

1102 symbolic names for datasets, like `ACS5` for `"acs/acs5" 

1103 in :py:module:`censusdis.datasets`. 

1104 vintage 

1105 The vintage to download data for. For most data sets this is 

1106 an integer year, for example, `2020`. But for 

1107 a timeseries data set, pass the string `'timeseries'`. 

1108 download_variables 

1109 The census variables to download, for example `["NAME", "B01001_001E"]`. 

1110 group 

1111 One or more groups (as defined by the U.S. Census for the data set) 

1112 whose variable values should be downloaded. These are in addition to 

1113 any specified in `download_variables`. 

1114 leaves_of_group 

1115 One or more groups (as defined by the U.S. Census for the data set) 

1116 whose leaf variable values should be downloaded.These are in addition to 

1117 any specified in `download_variables` or `group`. See 

1118 :py:meth:`VariableCache.group_leaves` for more details on the semantics of 

1119 leaves vs. non-leaf group variables. 

1120 set_to_nan 

1121 A list of values that should be set to NaN. Normally these are special 

1122 values that the U.S. Census API sometimes returns. If `True`, then all 

1123 values in :py:ref:`censusdis.values.ALL_SPECIAL_VALUES` will be replaced. 

1124 If `False`, no replacements will be made. 

1125 skip_annotations 

1126 If `True` try to filter out `group` or `leaves_of_group` variables that are 

1127 annotations rather than actual values. See :py:meth:`VariableCache.group_variables` 

1128 for more details. Variable names passed in `download_variables` are not 

1129 affected by this flag. 

1130 with_geometry 

1131 If `True` a :py:class:`gpd.GeoDataFrame` will be returned and each row 

1132 will have a geometry that is a cartographic boundary suitable for platting 

1133 a map. See https://www.census.gov/geographies/mapping-files/time-series/geo/cartographic-boundary.2020.html 

1134 for details of the shapefiles that will be downloaded on your behalf to 

1135 generate these boundaries. 

1136 remove_water 

1137 If `True` and if with_geometry=True, will query TIGER for AREAWATER shapefiles and 

1138 remove water areas from returned geometry. 

1139 api_key 

1140 An optional API key. If you don't have or don't use a key, the number 

1141 of calls you can make will be limited to 500 per day. 

1142 variable_cache 

1143 A cache of metadata about variables. 

1144 row_keys 

1145 An optional set of identifier keys to help merge together requests for more than the census API limit of 

1146 50 variables per query. These keys are useful for census datasets such as the Current Population Survey 

1147 where the geographic identifiers do not uniquely identify each row. 

1148 kwargs 

1149 A specification of the geometry that we want data for. For example, 

1150 `state = "*", county = "*"` will download county-level data for 

1151 the entire US. 

1152 

1153 Returns 

1154 ------- 

1155 A :py:class:`~pd.DataFrame` or `~gpd.GeoDataFrame` containing the requested US Census data. 

1156 """ 

1157 geos_kwargs = _intersecting_geos_kws( 

1158 dataset, vintage, self._containing_kwargs, **kwargs 

1159 ) 

1160 

1161 gdf = download( 

1162 dataset, 

1163 vintage, 

1164 download_variables, 

1165 group=group, 

1166 leaves_of_group=leaves_of_group, 

1167 set_to_nan=set_to_nan, 

1168 skip_annotations=skip_annotations, 

1169 with_geometry=True, 

1170 remove_water=remove_water, 

1171 api_key=api_key, 

1172 variable_cache=variable_cache, 

1173 row_keys=row_keys, 

1174 **geos_kwargs, 

1175 ) 

1176 

1177 # See which of these geometries are mostly contained by 

1178 # the geography we want to be within. 

1179 

1180 gdf_container = download( 

1181 dataset, vintage, ["NAME"], with_geometry=True, **self._containing_kwargs 

1182 ).drop("NAME", axis="columns") 

1183 

1184 gdf_contained = cmap.sjoin_mostly_contains( 

1185 gdf_container, gdf, area_threshold=self._area_threshold 

1186 ) 

1187 

1188 # Drop all the large container columns we don't need. 

1189 gdf_contained = gdf_contained[ 

1190 [col for col in gdf_contained.columns if not col.endswith("_large")] 

1191 ].reset_index(drop=True) 

1192 

1193 # Drop the "_small" suffix. 

1194 

1195 gdf_contained.rename( 

1196 lambda col: col[:-6] if col.endswith("_small") else col, 

1197 axis="columns", 

1198 inplace=True, 

1199 ) 

1200 

1201 if with_geometry: 

1202 # Keep the columns from the larger result. 

1203 return gdf_contained[ 

1204 [ 

1205 col 

1206 for col in gdf_container.columns 

1207 if col in gdf_contained.columns and col != "geometry" 

1208 ] 

1209 + [ 

1210 col 

1211 for col in gdf_contained.columns 

1212 if col not in gdf_container.columns or col == "geometry" 

1213 ] 

1214 ] 

1215 else: 

1216 # Drop the geometry and return a `pd.DataFrame` 

1217 return pd.DataFrame( 

1218 gdf_contained[ 

1219 [ 

1220 col 

1221 for col in gdf_container.columns 

1222 if col in gdf_contained.columns and col != "geometry" 

1223 ] 

1224 + [ 

1225 col 

1226 for col in gdf_contained.columns 

1227 if col not in gdf_container.columns and col != "geometry" 

1228 ] 

1229 ] 

1230 )
